\begin{enumerate}
\item Let $A$ and $B$ be sets. For each of the following propositions, either prove it or give a counterexample.
	\begin{enumerate}
	\item If $A \subseteq B$, then $\mathscr{P}(A) \subseteq \mathscr{P}(B)$.
	\item If $\mathscr{P}(A) \subseteq \mathscr{P}(B)$, then $A \subseteq B$.
	\end{enumerate} 

\item Prove that 
\[A \cup (B \cap C) = (A \cup B) \cap (A \cup C)\]
for all sets $A$, $B$, $C$.


\item Prove that for all sets $A$ and $B$, 
\[(A \cup B) - (A \cap B) = (A - B) \cup (B - A).\]


\item Let $A$ and $B$ be sets.
	\begin{enumerate}
	\item Prove that $\mathscr{P}(A \cap B) = \mathscr{P}(A) \cap \mathscr{P}(B)$.
	\item Is it true that $\mathscr{P}(A \cup B) = \mathscr{P}(A) \cup \mathscr{P}(B)$? Either prove or give a counterexample.
	\end{enumerate}


\item For each natural number $n$, let 
\[A_n = \left[\frac{1}{n^2}, 2n+1\right) = \left\{x \in \mathbb{R} : \frac{1}{n^2} \leq x < 2n + 1\right\}.\]
Calculate (with proof) $\displaystyle \bigcup_{n \in \mathbb{N}} A_n$ and $\displaystyle \bigcap_{n \in \mathbb{N}} A_n$.
\end{enumerate}
